# STAGE 1: Build the TypeScript/React application
# Use the Node.js image (version 18, alpine for lightness) as the "build environment"
FROM node:18-alpine AS builder

# Set the working directory inside the container
WORKDIR /app

# Copy package.json and package-lock.json files first
COPY package*.json ./

# Install dependencies
RUN npm install

# Copy the rest of the source code
COPY . .

# Run the build command (this will create the /app/dist directory)
RUN npm run build

# STAGE 2: Serve the built application with Nginx
# Use the Nginx image (alpine for lightness) as the "web server"
FROM nginx:alpine

# Copy ONLY the build directory (the result of Stage 1)
# into the default web root directory of Nginx
COPY --from=builder /app/dist /usr/share/nginx/html

# Copy the Nginx configuration file (already in your repo)
# This file is crucial for handling React routing (to avoid 404 errors on refresh)
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Expose port 80
EXPOSE 80

# Start Nginx
CMD ["nginx", "-g", "daemon off;"]
